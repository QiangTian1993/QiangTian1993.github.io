# 数组乱序

let arr = Object.keys(Array(100).fill())

function disorder(arr) {
	if (!Array.isArray(arr)) return

	for (let i = 0; i < arr.length; i++) {
		let newIndex = Math.floor(Math.random() * 100)
		let temp = arr[i]
		arr[i] = arr[newIndex]
		arr[newIndex] = temp
	}

	return arr
}

# 两数之和

给定一个整数数组 nums  和一个整数目标值 target，请你在该数组中找出和为目标值的那 两个 整数，并返回它们的数组下标。

思路：

  1. 利用 Map 字典
  2. 编辑 nums 数组，匹配符合信息的目标值 taget - currentValue
  3. 匹配成功返回两个数值下标
  4. 匹配失败将信息放到 Map 中
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
function toSum(nums, target) {
	const len = nums.length
	const MAP = new Map()
	for (let i = 0; i < len; i++) {
		const other = target - nums[i]
		if (MAP.has(other)) return [MAP.get(other), i]
		MAP.set(nums[i], i)
	}
}

# 大数相加

  1. number 类型有精度损失，只能是使用 string
  2.  转换成数组翻转，模拟加法运算
  3. 如果首位是 0 删除
/**
 *@param{string} num1
 *@param{string} num2
 *@return{string}
 */
function bigNumAdd(num1, num2) {
	const chukNum = num => typeof num == 'string' && !isNaN(Number(num))
	if (chukNum(num1) && chukNum(num2)) {
		const _num1 = num1.split('').reverse()
		const _num2 = num2.split('').reverse()
		const maxLen = Math.max(_num1.length, _num2.length)
		const format = num => {
			if (typeof num === 'number') return num
			if (!isNaN(Number(num))) return Number(num)
			return 0
		}
		let result = []
		let addTemp = 0
		for (let i = 0; i < maxLen; i++) {
			const addNum = format(_num1[i]) + format(_num2[i]) + addTemp
			addTemp = addNum > 9 ? 1 : 0
			result[i] = addNum % 10
		}
		result = result.reverse()
		const resultNum = result[0] > 0 ? result.join('') : result.join('').slice(1)
		console.log(resultNum)
		return result
	}
}
