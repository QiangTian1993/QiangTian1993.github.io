手写 Promise

IAN
2020-08-23 JavaScript ES6+ 
/**
 * promise 的状态是是不可逆的
 * PENDING -> FULFILLED
 * PENDING -> REJECT
 */
const PENDING = 'pending' // 等待
const FULFILLED = 'fulfilled' // 成功
const REJECTED = 'rejected' // 失败

class MyPromise {
	constructor(executor) {
		try {
			/**
			 * promise 接收一个执行器，这个执行器接收 resolve 和 reject 这两个函数
			 * resolve 会将 promise 的状态更改为 FULFILLED
			 * reject 会将 promise 的状态更改为 REJECTED
			 */
			executor(this.resolve, this.reject)
		} catch (err) {
			this.reject(err)
		}
	}

	/**
	 * promise 状态
	 */
	status = PENDING

	/**
	 * 用来保存 resolve 的值
	 */
	value = undefined

	/**
	 * 用来保存 reject 的原因
	 */
	reason = undefined

	/**
	 * 存放 then 中成功的回调函数
	 */
	successCallback = []

	/**
	 * 存放 then 中失败的回调函数
	 */
	failCallback = []

	/**
	 * @param value 成功状态的值
	 */
	resolve = value => {
		// 当 resolve 执行时如果状态是 pending 将状态变更为 fulfilled
		if (this.status !== PENDING) return
		this.status = FULFILLED
		// 将成功的值进行保存
		this.value = value
		// 循环 successCallback 队列
		while (this.successCallback.length) {
			// 执行回调函数并弹出队列
			this.successCallback.shift()(this.value)
		}
	}

	/**
	 * @param reason 失败的原因
	 */
	reject = reason => {
		if (this.status !== PENDING) return
		// 将 promise 状态变更为 REJECT
		this.status = REJECTED
		// 将失败的原因进行保存
		this.reason = reason
		// 循环 failCallback 队列
		while (this.failCallback.length) {
			// 执行回调函数并弹出队列
			this.failCallback.shift()(this.reason)
		}
	}

	/**
	 * promise 的 then 方法接收两个回调函数，第一个函数是成功的回调，第二个是失败的回调
	 * 当 promise 的状态为 FULFILLED 时执行 成功回调
	 * 当 promise 的状态为 REJECTED 时执行 失败回调
	 * then 方法会返回一个新的 promise
	 */
	then(successCallback, failCallback) {
		successCallback = successCallback ? successCallback : value => value
		failCallback = failCallback ? failCallback : value => value

		let promise = new MyPromise((resolve, reject) => {
			if (this.status === FULFILLED) {
				handlerSuccess.call(this)
			} else if (this.status === REJECTED) {
				handlerFail.call(this)
			} else {
				/**
				 * 如果执行到这里 promise 的状态还为 PENDING 说明 promise 接收的执行是异步的所以将 then 中的回调函数占时存储起来
				 * 将 then 中的回调函数存储在 successCallback 和 failCallback 中，当 promise 执行器调用 resolve 或 reject 来改变 promise 状态时再执行这些回调函数
				 */
				this.successCallback.push(() => {
					handlerSuccess.call(this)
				})
				this.failCallback.push(() => {
					handlerFail.call(this)
				})
			}

			function handlerSuccess() {
				// 使用定时器模拟 promise 的微任务
				return setTimeout(() => {
					try {
						// 执行成功回调函数并记录回调的返回结果
						let x = successCallback(this.value)
						// 如果是 successCallback  返回的是普通值就直接 resolve,
						// 如果是 promise 则主动调用 promise 的回调是 promise 状态传递下去
						// successCallback 返回的 promise 不能使自身
						resolvePromise(x, promise, resolve, reject)
					} catch (err) {
						reject(err)
					}
				}, 0)
			}
			function handlerFail() {
				setTimeout(() => {
					try {
						let x = failCallback(this.reason)
						resolvePromise(x, promise, resolve, reject)
					} catch (err) {
						reject(err)
					}
				}, 0)
			}
		})

		return promise
	}

	catch(failCallback) {
		return this.then(undefined, failCallback)
	}

	finally(callback) {
		return this.then(
			value => {
				return MyPromise.resolve(callback()).then(() => value)
			},
			reason => {
				return MyPromise.resolve(callback()).then(() => {
					throw reason
				})
			}
		)
	}

	static all(arr) {
		return new MyPromise((resolve, reject) => {
			const result = []
			let index = 0

			function addData(key, value) {
				index++
				result[key] = value
				if (index === arr.length) {
					resolve(result)
				}
			}

			for (let i = 0; i < arr.length; i++) {
				const cur = arr[i]
				if (cur instanceof MyPromise) {
					cur.then(
						value => {
							addData(i, value)
						},
						reason => {
							reject(reason)
						}
					)
				} else {
					addData(i, cur)
				}
			}
		})
	}

	static race(arr) {
		return new MyPromise((resolve, reject) => {
			for (let i = 0; i < arr.length; i++) {
				const cur = arr[i]
				if (cur instanceof MyPromise) {
					cur.then(resolve, reject)
				} else {
					return resolve(cur)
				}
			}
		})
	}
}

function resolvePromise(x, promise, resolve, reject) {
	if (promise == x) {
		return reject(new TypeError('Chaining cycle detected for promise #<Promise>'))
	}
	if (x instanceof MyPromise) {
		x.then(resolve, reject)
	} else {
		resolve(x)
	}
}
