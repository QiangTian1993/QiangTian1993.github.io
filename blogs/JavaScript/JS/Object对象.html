    # Object.defineProperty(),Object.defineProperties()

    Object.defineProperty 方法允许通过属性描述对象，定义或修改一个属性，然后返回修改后的对象

    # 属性描述对象

    属性描述对象是 js 内部定义的一个数据结构，用来描述一个对象的属性，控制它的行为，比如说该对象的某个属性是否是可被改写的，可被遍历的

    # Object.defineProperty() 的用法

    通过 Object.defineProperty 方法创建的属性默认不可改写，不可遍历

    let obj = {}
Object.defineProperty(obj, 'name', {
	value: '张三',
	writable: false,
	enumerable: false,
	configurable: false
})
console.log(obj) // {name: '张三'}
obj.name = '李四'
console.log(obj) // {name: '张三'}
Object.keys(obj) // {}

    writable 控制属性是否可被改写，不写默认是 false enumerable 控制属性是否可以被遍历，不写默认是 false configurable 控制描述对象属性的可写性，不写默认 false

    set、get 函数

    let obj = { name: 'tian', age: 18 }

for (let i in obj) {
	let val = obj[i] // 这里需要注意，需要一个变量来村寨
	Object.defineProperty(obj, i, {
		get() {
			return val
		},
		set(newVal) {
			val = newVal
		}
	})
}

    # Object.hasOwnProperty()

    for in 和 in 有个问题，它不仅能获取到对象中的属性，还能获取到原型中的属性。如果我们只想获取到对象中属性就可以用hasOwnProperty()

    function Obj(name, age) {
	this.name = name
	this.age = age
}
Obj.prototype.city = '北京'
let obj = new Obj('张三', 18)

for (let key in obj) {
	console.log(key) // name age city
}
'city' in obj //true
obj.hasOwnProperty('city') //false
