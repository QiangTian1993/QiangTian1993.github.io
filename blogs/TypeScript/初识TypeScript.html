                          # TypeScript 与 JavaScript 的区别

                          1. TypeScript 与 JavaScript 都是 ECMAScript 的实现
                          2. TypeScript 是静态类型， JavaScript 是动态类型
                          3. TypeScript 是 JavaScript 的超集
                          4. TypeScript 需要编译

                          # 枚举

                          enum EnumObj {
	'a' = 1,
	'b' = 2,
	'c' = 3,
	'd' = 4

console.log(EnumObj['a'], EnumObj[1])

                          # 泛型

                          定义接口或类的时候没有定义具体的类型，等到具体使用的时候再定义具体的类型，这种现象就叫做泛型

                          // 创建指定长度的数组，并用指定的值填充数据
function createArr<T>(length: number, value: T): Array<T> {
	let result: T[] = []
	for (let i = 0; i < length; i++) {
		result[i] = value
	}
	return result
}

createArr<number>(10, 1)

                          # 类的访问修饰符

                          class Person {
	readonly name: string // 只读属性不可修改
	private age: number // private 私有属性，不可被实例与子类访问
	public city: string // 共用属性，默认修饰符
	protected gender: string // 受保护的，不可被实例访问，与 private 的却别在于 protected 可以被子类访问

	constructor(name: string, age: number) {
		this.name = name
		this.age = age
		this.gender = '男'
	}
}

                          # 类与接口

                          interface Eat {
	eat(food: string): void
}
interface Run {
	run(distance: number): void
}

class Person implements Eat, Run {
	eat(food: string): void {
		console.log(`优雅的就餐 ${food}`)
	}
	run(distance: number): void {
		console.log(`直立行走 ${distance}`)
	}
}

class Animal implements Eat, Run {
	eat(food: string): void {
		console.log(`喂食 ${food}`)
	}
	run(distance: number): void {
		console.log(`爬行 ${distance}`)
	}
}

                          # type VS interface

                          1. type 可以声明基本类型、联合类型、元祖等
                          2. type 语句中还可以使用 typeof 获取实例的 类型进行赋值
                          3. interface 能够声明合并
                          4. type 不能 extends 和 implements

                          # 抽象类

                          abstract 关键字可可以定义抽象类，抽象类不可以直接使用 new 的方法创建实例，它只能被继承 abstract 也可以实现一个抽象方法，定义了抽象方法后，子类中必须实现此方法

                          abstract class Animal {
	eat(food: string): void {
		console.log(`喂食 ${food}`);
	}

	abstract run(distance: number): void
}

class Dog extends Animal {
	run(distance: number): void {
		console.log(`爬行 ${distance}`);
	}
}
